rec:
        # Prologue
        addi $sp, $sp, -8 # allocate space for $ra, $a0, $s0
        sw $ra, 4($sp) # save $ra at the top of stack
        
        # Body
        bne $a0, $0, calc_rec # if n == 0 goto base
        and $v0, $0, $0 # $v0 = 0, return value
        addi $t0, $0, 1 # get ready to turn on the first index of a
        la $t1, a
        sw $t0, 0($t1) # turn on said index - a[0] = 1;
        j rec_done

        # calculate rec(n - 1, a)
calc_rec:
	sw $a0, 0($sp) # need $a0 after jal
        addi $a0, $a0, -1 # $a0 = n - 1
        jal rec
        # Epilogue
        lw $ra, 4($sp) # load $ra
        lw $a0, 0($sp) # restore $a
        addi $sp, $sp, 8 # reduce the size of the stack

\end{verbatim}
